622 virtual std::unordered_set<
623 NodePtr, std::function<std::size_t(const NodePtr&)>,
624 std::function<bool(const NodePtr&, const NodePtr&)>>
625 Predecessor(const std::size_t& id) const {
626 decltype(_nodes) res(1, _nodes.hash_function(), _nodes.key_eq());
627
628 // Initialize the queue
629 auto first_parents = DiGraph<Node, Edge>::Parents(id);
630 res.merge(first_parents);
631
632 std::queue<NodePtr> q(first_parents.cbegin(), first_parents.cend());
633
634 // Add predecessor in a FIFO manner
635 while (!q.empty()) {
636 const auto n = q.front();
637
638 auto n_parents = DiGraph<Node, Edge>::Parents(n->Id());
639
640 for (auto& p : n_parents) {
641 if (!res.contains(p)) {
642 q.push(p);
643 }
644 }
645
646 res.merge(n_parents);
647 q.pop();
648 }
649
650 return res;
651 }
