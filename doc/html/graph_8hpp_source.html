628 virtual std::unordered_set<
629 NodePtr, std::function<std::size_t(const NodePtr&)>,
630 std::function<bool(const NodePtr&, const NodePtr&)>>
631 Predecessor(const std::size_t& id) const {
632 decltype(_nodes) res(1, _nodes.hash_function(), _nodes.key_eq());
633
634 // Initialize the queue
635 auto first_parents = DiGraph<Node, Edge>::Parents(id);
636 res.merge(first_parents);
637
638 std::queue<NodePtr> q(first_parents.cbegin(), first_parents.cend());
639
640 // Add predecessor in a FIFO manner
641 while (!q.empty()) {
642 const auto n = q.front();
643
644 auto n_parents = DiGraph<Node, Edge>::Parents(n->Id());
645
646 for (auto& p : n_parents) {
647 if (!res.contains(p)) {
648 q.push(p);
649 }
650 }
651
652 res.merge(n_parents);
653 q.pop();
654 }
655
656 return res;
657 }
